let content = '百度网站 \'www.baidu.com\'';
console.log(content);

let url = 'www.baidu.com';

# 常用方法

大小写转换

toUpperCase: 将字符转换成大写格式

console.log('baidu.com'.toUpperCase()); // BAIDU.COM
1

toLowerCase: 转字符为小写格式

console.log('BAIDU.COM'.toLowerCase()); //baidu.com
1

移除空白字符

trim: 移除两边空白字符 trimLeft：移除左边空白字符 trimRight: 移除右边空白字符

let str = '   baidu.com  ';

console.log(str.length); // 14
console.log(str.trim().length); // 9

获取单字符

  • charAt: 根据从0开始的位置获取字符
  • 使用数字索引获取字符串
let url = 'www.baidu.com'

console.log(url.charAt(4)) // b
console.log(url[4]) // b

截取字符串

使用 slice、substr、substring 函数都可以截取字符串

  • slice、substring 第二个参数为截取的结束位置
  • substr 第二个参数指定获取字符数量
let url = 'www.baidu.com';

url.slice(3); // .baidu.com
url.substr(3); // .baidu.com
url.substring(3); // .baidu.com

url.slice(3, 6); // .ba
url.substring(3, 6); // .ba
url.substring(3, 0); // www 较小的做为起始位置
url.substr(3, 6); // .baidu 第二个参数截取6个字符

url.slice(3, -1); // .baidu.co 第二个为负数表示从后面算的字符
url.slice(-2);// om 从末尾取
url.substring(3, -9); // www 负数转为0
url.substr(-3, 2); // co 从后面第三个开始取两个

查找字符串

  • indexOf: 从左往右开始获取字符串位置，检测不到时返回 -1
  • lastIndexOf: 相反，从右往左开始获取字符串位置，检测不到时返回 -1
'www.google.com'.indexOf('o'); // 5
'www.google.com'.indexOf('o', 7); // 12 从第7个字符向后搜索

'www.google.com'.lastIndexOf('o'); // 12
'www.google.com'.lastIndexOf('o', 7); // 6 从第7个字符向后搜索

  • startsWith 是否是指定位置开始，第二个参数为查找的开始位置

替换字符串

replace 方法用于字符串的替换操作

let name = "www.baidu.com";
let web = name.replace("baidu", "google");
console.log(web);
1
2
3

默认只替换一次，如果全局替换需要使用正则

let str = "2020/10/12";
console.log(str.replace(/\//g, "-"));
1
2

使用字符串替换来生成关键词链接

const word = ["www.baidu.com", "www.google.com"];
const string = "搜索引擎：www.baidu.com和www.google.com";
const title = word.reduce((pre, word) => {
  return pre.replace(word, `<a href="${word}">${word}</a>`);
}, string);

console.log(title) // "搜索引擎：<a href="www.baidu.com">www.baidu.com</a>和<a href="www.google.com">www.google.com</a>"

匹配字符

match: 匹配字符，返回数组类型，匹配不到返回 null, 默认匹配第一个，全局匹配返回多个结果

let url = 'www.baidu.com'

url.match(/w/) // ['w']
url.match(/s/) // null
url.match(/s/g) // ['w', 'w', 'w']
