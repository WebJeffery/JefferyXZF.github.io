# 数据类型

JavaScript 数据类型分为基本类型和引用类型

  • 基本类型: null, undefined, number, string, boolean, symbol, bigint
  • 引用类型: Object, Array, Function, Date, RegExp, Error 等等

# 基本类型 vs 引用类型

  • 存储方式：基本类型存储在栈内存中，引用类型存储在堆内存中，基本类型读写速度更快些
  • 拷贝方式：基本类型拷贝的是值的副本，修改值不会对原来的值有影响；引用类型拷贝的是指针的引用，修改对象属性值，会对原来的值有影响

# 类型检测

# typeof 检测

typeof 主要用于检测基本数据类型，它返回值是 string 类型字符串，结果值有以下几种

  • 基本类型：'undefined, 'number', 'string', 'boolean', 'symbol', 'bigint'
  • 引用类型：'function', 'object'

需要注意的是 typeof null 返回值是 'object', 而不是 'null'

🌰 代码事例

let a
console.log(typeof a); // 'undefined'

let b = null
console.log(typeof b); // 'object'

let c = 1
console.log(typeof c); // 'number'

let d = '1'
console.log(typeof d); // 'string'

let e = true
console.log(typeof e); // 'boolean'

let f = Symbol(1)
console.log(typeof f); // 'symbol'

let g = BigInt(1)
console.log(typeof g); // 'bigint'

let arr = []
console.log(typeof arr); // 'object'

let obj = {}
console.log(typeof obj); // 'object'

let fn = function () {}
console.log(typeof fn); // 'function'

let reg = new RegExp(/\d/, 'i')
console.log(typeof reg); // 'object'

let date = new Date()
console.log(typeof date); // 'object'

// 基本数据类型构造器

let num = new Number() // NUmber Boolean String
console.log(typeof num); // 'object'

# instanceof 检测

instanceof 运算符用于检测构造函数的 prototype 属性是否出现在某个实例对象的原型链上

typeof 不能区分数组，但 instanceof 可以, 也可以使用 Array.isArray

🌰 代码事例

let arr = [];
let obj = {};
console.log(arr instanceof Array); // true
console.log(obj instanceof Array); // false

let c = [1, 2, 3];
console.log(c instanceof Array); //true

let d = { name: "zhangsan" };
console.log(d instanceof Object); //true

function User() {}
let user = new User();
console.log(user instanceof User); // true 等同于 user.__proto__.constructor === User

# toString 方法检测

toString 方法返回的是字符串类型 '[object 构造函数类型]',

返回值有以下，[object Undefined], [object Null], [object Number], [object String], [object Boolean], [object Symbol], [object BigInt], [object Array], [object Object], [object Function], [object RegExp], [object Date] 等等

let toStrFun = Object.prototype.toString

let a
console.log(toStrFun.call(a)); // '[object Undefined]'

let b = null
console.log(toStrFun.call(b)); // '[object Null]'

let c = 1
console.log(toStrFun.call(c)); // '[object Number]'

let d = '1'
console.log(toStrFun.call(d)); // '[object String]'

let e = true
console.log(toStrFun.call(e)); // '[object Boolean]'

let f = Symbol(1)
console.log(toStrFun.call(f)); // '[object Symbol]'

let g = BigInt(1)
console.log(toStrFun.call(g)); // '[object BigInt]'

let arr = []
console.log(toStrFun.call(arr)); // '[object Array]'

let obj = {}
console.log(toStrFun.call(obj)); // '[object Object]'

let fn = function () {}
console.log(toStrFun.call(fn)); // '[object Function]'

let reg = new RegExp(/\d/, 'i')
console.log(toStrFun.call(reg)); // '[object RegExp]'

let date = new Date()
console.log(toStrFun.call(date)); // '[object Date]'

function User () {}
let user = new User()
console.log(toStrFun.call(user)); // '[object Object]'

// 基本数据类型构造器

let num = new Number() // NUmber Boolean String
console.log(toStrFun.call(num)); // '[object Number]'

# 封装数据类型检测方法

根据 typeof， instanceof, toString 数据类型检测方式，使用 toString 检测更通用些，可以封装成方法

const isType = (target) => {
  return type => {
    return Object.prototype.toString.call(type) === `[object ${target}]`
  }
}

const isNumber = isType('Number')
const isString = isType('String')
const isBoolean = isType('Boolean')
const isArray = isType('Array')
const isFunction = isType('Function')
const isObject = isType('Object')
const isRegExp = isType('RegExp')

console.log(isNumber(22)) // true
console.log(isNumber('22')) // false
console.log(isArray([22])) // true
console.log(isArray({a: 33})) // false
console.log(isObject({a: 33})) // true
